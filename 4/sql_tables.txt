CREATE TABLE ADVERTISER (
    PRIMARY KEY ID INT NOT NULL UNIQUE,
    NAME VARCHAR(256) NOT NULL,
    CLICKS INT,
    VIEWS
 );

 CREATE TABLE AD (
    PRIMARY KEY ID NOT NULL UNIQUE,
    TITLE VARCHAR(256) NOT NULL,
    IMGURL VARCHAR(256),
    LINK VARCHAR(256).
    FOREIGN KEY ADV REFERENCES ADVERTISER.ID
 );
